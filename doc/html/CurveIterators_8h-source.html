00190       {
00191         // we're setting the iterator to end
00192         ++__A;
00193         ++__B;

00216       float normAB = (float)vec_tmp2.norm();
00217

00219       {
00220         _CurvilinearLength += _step;

00225         //if normAB ~= 0, we don't change these values

00280       }
